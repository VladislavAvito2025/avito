const type = document.getElementById("type").options[document.getElementById("type").selectedIndex].text; const width = document.getElementById("width").value; const length = document.getElementById("length").value; const frame = document.getElementById("frame").options[document.getElementById("frame").selectedIndex].text; const polycarbonate = document.getElementById("polycarbonate").options[document.getElementById("polycarbonate").selectedIndex].text; const arcStep = document.getElementById("arc-step").value || "1 м"; const foundation = document.getElementById("foundation").options[document.getElementById("foundation").selectedIndex].text; const delivery = document.getElementById("delivery").value || 0; // Получаем значение доставки // Рассчитываем общую стоимость const totalCost = parseFloat(document.getElementById('type').value) + (width * length * parseFloat(document.getElementById('frame').value)) + (width * length * parseFloat(document.getElementById('polycarbonate').value)) + parseFloat(document.getElementById('foundation').value) + (delivery * 50); // Предположим, что стоимость доставки 50 руб/км // Формируем текст для всплывающего сообщения const popupContent = ` Тип теплицы: ${type} ☀️ Поликарбонат с защитой от ультрафиолета UV-400 ⚙️ Цельногнутые дуги из стальной замкнутой трубы ⚙️ Каркас из оцинкованной внутри и снаружи стали (полностью защита от ржавчины и коррозии) ⚙️ Краб-система на 4 болтах ❄️ Не требует подпорок на зиму ✏️ Гарантия 15 лет!!! ----------------- Каркас: ${frame} Шаг дуги: ${arcStep} Поликарбонат: ${polycarbonate} с защитой от ультрафиолета Размеры (ШхД): ${width} x ${length} м² Комплектация: 2 двери, 2 форточки Фундамент: ${foundation} Доставка: ${delivery} км (${delivery * 50} руб) ----------------- Стоимость со скидкой до ${getDiscountDate()}: ${totalCost.toFixed(2)} руб. (в стоимость включены: чек, каркас на краб-системе, поликарбонат, инструкция по сборке) `; // Отображаем всплывающее сообщение document.getElementById("popup-content").innerText = popupContent; document.getElementById("overlay").style.display = "block"; document.getElementById("popup").style.display = "block"; } // Функция для получения даты скидки (текущая дата + 7 дней) function getDiscountDate() { const date = new Date(); date.setDate(date.getDate() + 7); return date.toLocaleDateString("ru-RU"); } // Копирование текста в буфер обмена function copyToClipboard() { const text = document.getElementById("popup-content").innerText; navigator.clipboard.writeText(text).then(() => { alert("Текст скопирован!"); }); } // Закрытие всплывающего сообщения при клике на оверлей document.getElementById("overlay").addEventListener("click", () => { document.getElementById("overlay").style.display = "none"; document.getElementById("popup").style.display = "none"; });
